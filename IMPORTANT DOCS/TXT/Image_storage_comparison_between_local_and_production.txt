public function getThumbnailUrlAttribute()
    {
        if (!$this->thumbnail_path) {
            return null;
        }
        return Storage::disk('public')->exists($this->thumbnail_path)
            ? asset('storage/' . $this->thumbnail_path)
            : null;
    }
